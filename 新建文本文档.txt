        // 解析自定义单词
        function parseCustomWords(text) {
            const lines = text.split('\n');
            const result = [];
            
            for (const line of lines) {
                if (!line.trim()) continue;
                const [word, meaning] = line.split('-');
                if (word && meaning) {
                    result.push({
                        word: word.trim(),
                        meaning: meaning.trim()
                    });
                }
            }
            
            return result;
        }

        // 从单词库中随机选择题目
        function getRandomQuestions(wordList, count) {
            const shuffled = [...wordList].sort(() => 0.5 - Math.random());
            return shuffled.slice(0, count);
        }

        // 显示问题
        function showQuestion() {
            if(currentQuestion >= questions.length) {
                endGame();
                return;
            }
            
            const q = questions[currentQuestion];
            currentWordEl.textContent = q.word;
            wordMeaningEl.textContent = q.meaning;
            wordMeaningEl.style.opacity = '0';
            
            progressText.textContent = `${currentQuestion+1}/${totalQuestions}`;
            
            // 生成选项
            optionsContainer.innerHTML = '';
            
            // 正确答案
            const correctOption = document.createElement('div');
            correctOption.className = 'option-btn';
            correctOption.textContent = q.meaning;
            correctOption.addEventListener('click', () => checkAnswer(true, q.meaning, q.word));
            optionsContainer.appendChild(correctOption);
            
            // 错误选项(从其他单词中随机选取)
            const otherMeanings = wordDatabase[currentLevel]
                .filter(item => item.meaning !== q.meaning)
                .map(item => item.meaning);
            
            // 随机选择3个错误选项
            for(let i = 0; i < 3; i++) {
                const randomIndex = Math.floor(Math.random() * otherMeanings.length);
                const wrongOption = document.createElement('div');
                wrongOption.className = 'option-btn';
                wrongOption.textContent = otherMeanings[randomIndex];
                wrongOption.addEventListener('click', () => checkAnswer(false, q.meaning, q.word));
                optionsContainer.appendChild(wrongOption);
                
                // 避免重复
                otherMeanings.splice(randomIndex, 1);
            }
            
            // 打乱选项顺序
            for (let i = optionsContainer.children.length; i >= 0; i--) {
                optionsContainer.appendChild(optionsContainer.children[Math.random() * i | 0]);
            }
            
            // 隐藏反馈和下一题按钮
            correctFeedback.style.display = 'none';
            incorrectFeedback.style.display = 'none';
            nextBtn.classList.add('hidden');
        }

        // 检查答案
        function checkAnswer(isCorrect, correctAnswer, currentWord) {
            // 禁用所有选项
            const allOptions = document.querySelectorAll('.option-btn');
            allOptions.forEach(opt => {
                opt.style.pointerEvents = 'none';
                if(opt.textContent === correctAnswer) {
                    setTimeout(() => {
                        opt.classList.add('correct');
                    }, 300);
                }
            });
            
            if(isCorrect) {
                score++;
                correctFeedback.style.display = 'block';
                // 显示单词释义
                wordMeaningEl.style.opacity = '1';
            } else {
                incorrectFeedback.style.display = 'block';
                correctAnswerEl.textContent = correctAnswer;
                // 显示单词释义
                wordMeaningEl.style.opacity = '1';
                
                // 记录错误单词
                const wrongWord = questions[currentQuestion];
                wrongWords.push(wrongWord);
            }
            
            // 更新学习进度
            wordsLearnedToday++;
            updateDailyProgress();
            
            nextBtn.classList.remove('hidden');
        }
